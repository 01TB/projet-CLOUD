# ---------------- lancement du projet ----------------
# 1. Rendre exécutable le script d'initialisation
chmod +x init-project.sh

# 2. Lancer l'initialisation (tout sera configuré automatiquement)
./init-project.sh

# 3. Alternative: Lancer manuellement
docker-compose up --build -d

# 4. Vérifier que tout fonctionne
docker-compose ps
curl http://localhost:8080/api/health
curl http://localhost:8081/
